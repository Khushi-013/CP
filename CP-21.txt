/*Function call by value is a method of passing arguments to the Function
When we pass a variable by value it makes a copy of the variable
Changes made to the arguments nside the function will remain in the function
It will not be affected outside the function will remain in the function*/
#include<stdio.h>
int cube(int x)
{
    x=x*x*x;
    return(x);
}
int main()
{
    int n=8;
    printf("Cube of %d is %d\n",n,cube(n));
    return 0;
}
